	\subsection{Atributos de Calidad y Concerns}
		\begin{frame}[allowframebreaks,from second]
			\frametitle{\insertsubsection}
			Los \textbf{atributos de calidad} (QAs) hablan de características específicas que debe tener el sistema, antes (mal)
			llamados ``requerimientos no funcionales''.\\[10pt]
			Ejemplo: eficiencia, disponibilidad, flexibilidad, usabilidad, etc.\\[16pt]

			\nota{ejemplo de definición QA: el sistema debe ser diseñado de manera tal que se minimize el procesamiento y el tiempo de respuesta.}

			Los \textbf{\emph{concerns}} permiten juzgar, especificar y medir los atributos de calidad. Usualmente, los requerimientos
			de atributos de calidad son expresados en términos de \emph{concerns}.
		
			\framebreak

			\nota{las invariantes seran utilizadas por Rainbow para saber cuándo y cómo debe adaptar el sistema}

			\scriptsize
			\begin{Verbatim}[gobble=4]
				Property MAX_RESPTIME : float = 1000.0;

				Component s1 : ServerT {
				  Property deploymentLocation;
				  Port http0;
				  Property cost;
				  Property fidelity;
				  Property load;
				}

				Component c1 : ClientT {
				  Property experRespTime : float << default : float = 100.0; >> ;
				  Property requestRate;
				  rule primaryConstraint = invariant self.experRespTime <= MAX_RESPTIME;
				}
			\end{Verbatim}

		\begin{frame}
			\frametitle{Tácticas y Estrategias de Reparación}

			\underline{\textbf{Táctica}}: decisión de diseño para controlar las respuestas del sistema a determinados
			estímulos, a fin de satisfacer uno o más QA.\\[5pt]
			e.g. \alert{agregar un servidor} al sistema.\\[14pt]
			\underline{\textbf{Estrategia}}: es un \textbf{algoritmo que usa una o más tácticas} para cumplir los
			atributos de calidad en un nivel deseado. 
		\end{frame}

		\begin{frame}
			\frametitle{Ejemplo}

			\textbf{Invariante}: el tiempo de respuesta experimentado por el usuario no debe superar el umbral
			determinado.\\[10pt]
			
			\nota{explicar disminuir fidelidad de la información: por ejemplo eliminar videos}
			
			\textbf{Estrategia} que intenta reparar el sistema en caso de romperse el invariante:\\[5pt]
			\begin{enumerate}
			  \item \alert{agregar un servidor}
			  \item si tiempo de respuesta $>$ umbral determinado
			  \textbf{\textcolor{blue}{$\Rightarrow$}} \alert{disminuir la fidelidad}
			  \item Estrategia exitosa \textbf{\textcolor{blue}{$\iff$}} tiempo de respuesta $<$ umbral determinado  
			\end{enumerate}
		\end{frame}

			\begin{frame}
				\frametitle{\insertsubsubsection}
				
				Arco Iris pretende involucrar a los \emph{stakeholders} en el proceso de auto reparación y proveerles

			\end{frame}

			\begin{frame}[allowframebreaks,from second]
				\frametitle{SHS: Extensión a QAS}

				\nota{Explicar importancia de ESTIMULO, ENTORNO y RESPONSE MEASURE para la auto adaptación.}

				Arco Iris define una extensión de los QAS denominada \textbf{Self Healing Scenario (SHS)}, agregando:
			  	\begin{itemize}
			    	\item Cada escenario tiene una prioridad determinada
			    	\item Cada entorno define la importancia de los distintos \emph{concerns}
			    	\item Estrategias que reparan al escenario
				\end{itemize}

				\begin{center}
					Arco Iris = Rainbow + Self Healing Scenarios
				\end{center}

				\begin{itemize}
					\item Permite asignar una medida relativa de satisfacción (i.e. utilidad) sobre un sistema sobre el cual es
					necesario medir el impacto ante determinados cambios.
					\item 100\% de utilidad cuando:
					\begin{itemize}
						\item no se viola ninguna restricción impuesta sobre el modelo (\textbf{Rainbow})
						\item todos los escenarios definidos se cumplen (\textbf{Arco Iris})
					\end{itemize}
				\end{itemize}

				\begin{Verbatim}[fontsize=\tiny,gobble=5]
					utilities:
					  uR:
					    label:       Average Response Time
					    mapping:     "[EAvg]ClientT.experRespTime"
					    description: "Client experienced response time in milliseconds, R, defined
					                  as a float property 'ClientT.experRespTime' in the architecture"
					    utility:
					      0:    1.00
					      500:  0.90
					      1000: 0.75
					      1500: 0.50
					      2000: 0.25
					      4000: 0.00
					  uF:
					    label:        Average Fidelity
					    mapping:      "[EAvg]ServerT.fidelity"
						...
					vectors:         # Tactic quality attribute vectors
					  enlistServers:
					    uR: -144
					    uF: 0
					    uC: +1.00
					...
				\end{Verbatim}
			\end{frame}

			\begin{frame}[containsverbatim]
				\frametitle{Puntuación y Selección de una Estrategia}
		
					\textbf{Puntuación}: predicción de la utilidad del sistema luego de simular la ejecución de la estrategia. 
				
					\begin{Verbatim}[fontsize=\tiny,gobble=6]
						scoreStrategia = 0
						estimacionPorConcern = simular la aplicación de la estrategia como lo hace Rainbow y
						                       obtener el valor resultante para los concerns
						Por cada escenario habilitado
							escenarioSeSatisface = !escenario.applyFor(entornoActual) o
							   (escenario.applyFor(entornoActual) y escenario.responseMeasure.holds(estimacionPorConcern))

							Si (escenarioSeSatisface)
								prioridadRelativa = calcular prioridad relativa del escenario
								pesoConcern = peso que el entorno actual asigna a escenario.concern
								puntajeEscenario = prioridadRelativa * pesoConcern
								scoreStrategia = scoreStrategia + puntajeEscenario
							Fin
						Fin
					\end{Verbatim}
					\textbf{Selección}: se elige aquella estrategia que posea el puntaje máximo.
			\end{frame}
