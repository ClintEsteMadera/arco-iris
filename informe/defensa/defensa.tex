	\subsection{Atributos de Calidad y Concerns}
		%--------- JONY ---------
		\begin{frame}[allowframebreaks]
			\frametitle{\insertsubsection}

			\nota{ejemplo de definición QA: el sistema debe ser diseñado de manera tal que se minimize el procesamiento y el tiempo de respuesta.}
			\nota{(mal) llamados requerimientos no funcionales $\Rightarrow$ Porque a veces se necesita implementar funcionalidad para
			ellos. e.g. seguridad: autorización requiere que se implemente funcionalidad}

			Los \textbf{atributos de calidad} (QAs) hablan de características específicas que debe tener el sistema, antes (mal)
			llamados ``requerimientos no funcionales''.\\[10pt]

			Los \textbf{\emph{concerns}} permiten juzgar, especificar y medir los atributos de calidad. Usualmente, los requerimientos
			de atributos de calidad son expresados en términos de \emph{concerns}.
			
			\begin{center}
				\textbf{Ejemplos}
				\small
				\rowcolors*[\hline]{1}{ForestGreen!55}{ForestGreen!20}
				\begin{tabular}{|c|c|}
					\textbf{Atributo de Calidad} & \textbf{Ejemplos de \emph{Concerns}}\\
					Eficiencia & Comportamiento Temporal, Utilización de Recursos\\
					Funcionalidad & Interoperabilidad, Seguridad\\
					Mantenibilidad & Cambiabilidad, Facilidad de Prueba\\
					Portabilidad & Adaptabilidad, Coexistencia\\
					Usabilidad & Compresibilidad, Atractivo			
				\end{tabular}
			\end{center}
		\end{frame}

	\subsection{Rainbow}

		%--------- TURCO ---------
		\begin{frame}
			\frametitle{\insertsubsection}
			
			\nota{Adapt.Basada en modelos de Arq.: módulo q' conoce e interpreta el modelo de la arq. y de acuerdo al
			problema detectado, toma una decisión sobre cómo reparar al sistema}
			\nota{Intenta aplicar a arquitecturas disímiles}
			\nota{No requiere diseñar el sistema teniendo en cuenta que va a ser adaptado mediante Rainbow}
			
			\begin{itemize}[<+->]
			  	\item \emph{Framework} de Auto Adaptación.
			  	\item Realizado en el marco del proyecto ABLE de la CMU. 
				\item Trabaja con el modelo de la arquitectura (expresado mediante \textbf{Acme}).
				\item Externo al sistema a adaptar, no intrusivo.
				\item Soporta sistemas \emph{legacy} y/o cuyo código no se encuentra disponible.
			\end{itemize}
			
		\end{frame}

		%--------- TURCO ---------
		\begin{frame}[containsverbatim]
			\frametitle{Ejemplo de una Arquitectura en Acme}
			
			\nota{los invariantes seran utilizados por Rainbow para saber cuándo y cómo debe reparar el sistema utilizando
			estrategias, las cuales seran explicadas mas adelante}
			\nota{Decir ``Tiempo de respuesta experimentado por el usuario''}
			\nota{Relacionar experRespTime con el concern Comportamiento Temporal para que se entienda ``Puntuación y
			Selección de una Estrategia''}

			\scriptsize
			\begin{Verbatim}[gobble=4]
				Property MAX_RESPTIME : float = 1000.0;

				Component s1 : Server {
				  Property deploymentLocation;
				  Port http0;
				  Property fidelity;
				  Property load;
				}

				Component c1 : Client {
				  Property experRespTime : float << default : float = 100.0; >> ;
				  Property requestRate;
				  rule primaryConstraint = invariant self.experRespTime <= MAX_RESPTIME;
				}
			\end{Verbatim}
		\end{frame}

		%--------- TURCO ---------
		\begin{frame}
			\frametitle{Tácticas y Estrategias de Reparación}

			\nota{Tanto Rainbow como Arco Iris utilizan los conceptos táctica y estrategia para reparar al sistema}

			\underline{\textbf{Táctica}}: decisión de diseño que permite modificar el comportamiento del sistema ante 
			determinados estímulos.\\[5pt] 
			e.g. \alert{agregar un servidor} al sistema.\\[14pt]
			\underline{\textbf{Estrategia}}: es un \textbf{algoritmo que usa una o más tácticas} para cumplir los
			atributos de calidad en un nivel deseado. 
		\end{frame}

		%--------- TURCO ---------
		\begin{frame}
			\frametitle{Ejemplo}

			\textbf{Invariante}: el tiempo de respuesta experimentado por el usuario no debe superar el umbral
			determinado.\\[10pt]
			
			\nota{Mencionar tiempo de asentamiento de una tactica}
			\nota{explicar disminuir fidelidad de la información: por ejemplo eliminar videos}
			\nota{La estrategia informa si pudo reparar el problema o no (Retorna SUCCESS o FAILURE)}
			
			\textbf{Estrategia} que intenta reparar el sistema en caso de romperse el invariante:\\[5pt]
			\begin{enumerate}
			  \item \alert{agregar un servidor}
			  \item si tiempo de respuesta $>$ umbral determinado
			  \textbf{\textcolor{blue}{$\Rightarrow$}} \alert{disminuir la fidelidad}
			  \item Estrategia exitosa \textbf{\textcolor{blue}{$\iff$}} tiempo de respuesta $<$ umbral determinado  
			\end{enumerate}
		\end{frame}

				\frametitle{SHS: Extensión a QAS}

				Arco Iris define una extensión de los QAS denominada \textbf{Self Healing Scenario (SHS)}, agregando:
			  	\begin{itemize}
			    	\item Cada escenario tiene una prioridad determinada
			    	\item Estrategias que reparan al escenario
			    	\item Cada entorno define la importancia de los distintos \emph{concerns}.\\[10pt]
			    	Por ejemplo, en un entorno de ``Alta Carga'':\\[10pt]
			    	\begin{tabular}{rcl}
				    	Tiempo de Respuesta&=&60\%\\
