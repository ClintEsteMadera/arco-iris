\section{Objetivos del Proyecto}
	\begin{frame}
		\frametitle{Objetivos del Proyecto}
		\scriptsize\begin{itemize}
			\item Mostrar la conveniencia y factibilidad del uso de un \textbf{método iterativo} de estimación que se base en la
			\textbf{tipificación} de los elementos del proyecto, de manera de:
				\begin{itemize}
					\item \scriptsize Permitir \textbf{reutilizar} los componentes de estimación a lo largo de los diferentes
					desarrollos.
					\item \scriptsize Brindar medidas adecuadas en cada futura estimación.
				\end{itemize}
			\item Proponer una \textbf{herramienta} que \textbf{implemente} esta técnica de estimación y sirva de ayuda gerencial
			a los lideres de proyectos para:
				\begin{itemize}
					\item \scriptsize la \textbf{planificación, estimación, especificación y actualización} constante de los proyectos.
					\item \scriptsize ser un \textbf{repositorio} de la información relevante de proyectos, accesibles por todos los
					recursos involucrados.
				\end{itemize}
		\end{itemize}
	\end{frame}

\section{Método Propuesto}
	\subsection{Justificación}
		% Enunciado
		\begin{frame}
			\frametitle{Estimación de Proyectos}
			``Su objetivo es conocer en etapas tempranas y de manera aproximada, el tamaño, esfuerzo y costo del desarrollo de un
			proyecto de software''
		\end{frame}
		
		\begin{frame}
			\frametitle{Métodos Tradicionales}
			\scriptsize	\begin{itemize}
				\item  \textbf{Métodos Empiricos}\\Dependen exclusivamente del juicio experto.
				\item  \textbf{Líneas de Código}
					\begin{itemize} \scriptsize	
						\item Qué es una línea de código?
						\item Desarrollo de sistema $\neq$ codificación
					\end{itemize}
					Dependiente de la tecnología, lenguajes, herramientas o metodologías utilizadas.
				\item  \textbf{Puntos de Función}\\
					Allan Albrecht - 1979: cuantifica el tamaño y la complejidad de un sistema de software según las funciones de
					usuario. No es útil en etapas tempranas (datos y funciones no están bien definidos)
				\item  \textbf{Puntos de Casos de Uso}\\
					Gustav Karner - 1993: clasifica los CU y actores en complejos, medios y simples.
					No considera cuestiones independientes de la funcionalidad.
				\end{itemize}
				No consideran en su aplicación la \textcolor{red}{historia} de los proyectos anteriormente realizados en la
				organización.
		\end{frame}

	\subsection{Método propuesto}
		\begin{frame}
			\frametitle{Elementos de Estimación}
			\scriptsize La tipificación de estos componentes por características comunes la definen los estimadores en base a:
				\begin{itemize}
					\item experiencia
					\item entorno organizacional
					\item requerimientos
				\end{itemize}
		\end{frame}

		\begin{frame}
			\frametitle{Método propuesto}
			\underline{Características}
			\scriptsize
			\begin{itemize}
				\item Adaptable fácilmente a las características propias de la organización 
				\item Considera todos los procesos involucrados en un desarrollo típicos de la cultura organizacional
				\item Reutiliza la definición de tipos de componentes a lo largo de diferentes proyectos.
				\item Con el uso las estimaciones son más refinadas y ajustadas a la realidad.
				\item \textbf{Método Iterativo}
					\begin{itemize}
						\scriptsize
						\item Iteración: realización completa de un proyecto.
						\item Comienza con valores más o menos aproximados.
						\item A medida que itere sus valores son retroalimentados y ajustados
						\item Con cada iteración se tiene una definición de tipos más extensa.
					\end{itemize}
			\end{itemize}
		\end{frame}

		\begin{frame}
			\frametitle{Implementación del Método}
			Pasos:
			\pause
			\scriptsize
			\begin{itemize}
				\item Tipificar los componentes del proyecto
					\begin{itemize}
						\item \scriptsize Esfuerzo de desarrollo
						\item \scriptsize Tiempo de desarrollo
						\item \scriptsize Recursos empleados
					\end{itemize}
				\pause
				\item Realizar la planificación de tareas que compondrá al proyecto y definir un tipo para cada una de las mismas.
				\pause
				\item Finalizada la primera iteración del método los valores para cada tipo de componente serán recalculados
				utilizando los resultados reales.
			\end{itemize}
		\end{frame}	

		\begin{frame}
			\frametitle{Ventajas del Método}
			\scriptsize
			\begin{itemize}
				\item Es un método relativamente fácil y familiar para los estimadores, lo cual \textbf{favorece su utilización}.
				\item Después de algunas iteraciones, la mayoría de las tareas ya se encuentran tipificadas, lo cual
				\textbf{facilita la planificación}.
				\item Permite \textbf{usar toda la información histórica} que generalmente se pierde.
				\item Luego de algunas iteraciones \textbf{el método se adapta a la organización} con resultados más precisos.
				\item Captura el conocimiento de los especialistas estimadores (\textbf{independencia de las personas}), previendo
				que dichos especialistas puedan desvincularse de la organización.
			\end{itemize}
		\end{frame}	
			
		\begin{frame}
			\frametitle{Desventajas del Método}
			\scriptsize
			\begin{itemize}
				\item Poca certeza en las primeras iteraciones del mismo, poseyendo la misma incertidumbre de los métodos utilizados
				para efectuar las primeras aproximaciones.
				\item No es directamente aplicable a las empresas de desarrollo de software para terceros (e.g. \emph{Software
				Factories}), ya que en general los proyectos suelen ser muy diferentes uno del otro, y es más difícil establecer
				características en común.
				\item Requiere la participación activa de los \emph{stakeholders} del proyecto, informando el desarrollo real de sus
				tareas, lo cual no siempre es fácil de conseguir.
			\end{itemize}
		\end{frame}	

\section{Resources Manager}
	\subsection{Descripción}
		\begin{frame}
			\frametitle{Características de Resources Manager}
			\scriptsize
			\begin{itemize}
				\item Automatiza el \textcolor{red}{método} propuesto
				\pause
				\item Permite la administración de \textcolor{red}{proyectos} y la gestión de \textcolor{red}{recursos},
				\textcolor{red}{tareas}, alertas y desvíos
				\pause
				\item Se \textcolor{red}{adapta} a las características de la organización.
				\pause
				\item Facilita la \textcolor{red}{planificación} de un proyecto
				\pause
				\item De \textcolor{red}{fácil} uso y acceso
				\pause
				\item \textcolor{red}{Repositorio} de la información valiosa para la organización, a utilizar no sólo en
				estimaciones sino también en otros aspectos clave como por ejemplo la mejora continua de procesos.
				\pause
				\item Permite la definición de \textcolor{red}{métricas} específicas a medir.
				\pause
				\item Refleja las \textcolor{red}{actividades} diarias de los recursos
				\pause
				\item Ofrece información de avance y sumarizada de los proyectos.
			\end{itemize}
		\end{frame}

	\subsection{Tipos de Componentes Resultantes}
		\begin{frame}
			\frametitle{Primera Iteración}
			\scriptsize Al incorporarse los datos reales aportados por Emisores, la información de tipos de componentes se actualizó de acuerdo a la siguiente fórmula:
			\begin{center}
				\begin{displaymath}
					TC_{n}= \frac{(\sum_{i=1}^{m} CH)+ TC_{n-1} + VI}{m+2}
				\end{displaymath}
			\end{center}
			donde
			$n$ = Número de Iteración\\
			$m$ = Cantidad de Componentes en el Historial de ese Tipo \\
			$TC$ = Tipo de Componente\\
			$CH$ = Componente del Historial\\
			$VI$ = Valor Ingresado\\
		\end{frame}
