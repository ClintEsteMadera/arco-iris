%los \Escenarios de Atributos de Calidad" como

	\subsubsection{Escenarios de Atributos de Calidad}
		\begin{frame}[allowframebreaks,from second]
			\frametitle{\insertsubsubsection}
			Los Escenarios de Atributos de Calidad o QAS(\emph{Quality Attribute Scenarios}) permiten expresar cómo debería
			responder el sistema ante determinados estímulos y bajo ciertas circunstancias.\\
			
			\framebreak
			
			\frametitle{\insertsubsubsection}
			Composición de un QAS:
			\begin{itemize}
				\item \textbf{Fuente del estímulo}: entidad que generó el estímulo
				\item \textbf{Estímulo}: condición que debe ser tenida en cuenta cuando llega al sistema
				\item \textbf{Entorno}: condiciones en las que se encuentra el sistema, e.g. alta carga
				\item \textbf{Artefacto}: el sistema o partes de él afectadas por el estímulo
				\item \textbf{Respuesta}: qué hace el sistema ante la llegada del estímulo
				\item \textbf{Cuantificación de la respuesta}: métrica sobre la \textbf{respuesta}
			\end{itemize}

			La \textbf{Cuantificación de la respuesta} permitirá saber si el escenario se satisface.
		\end{frame}
	
	\subsubsection{Quality Attribute Workshop (QAW)}
		\begin{frame}
			\frametitle{\insertsubsubsection}
			Quality Attribute Workshop es un método que define un proceso que involucra a los \emph{stakeholders} en la
			definición de los Escenarios de Atributos de Calidad.
			
			Permite identificar drivers de la Arquitectura del sistema
		\end{frame}

	\subsection{Arco Iris}
	
		\subsubsection{¿Qué es Arco Iris?}
	
			\begin{frame}
				\frametitle{\insertsubsubsection}
				Arco Iris pretende involucrar a los \emph{stakeholders} en el proceso de auto reparación y proveerles
				una mayor visibilidad de sus posibles consecuencias para el desempeño del sistema.
	
				\begin{itemize}
				  \item Permite a stakeholders definir los requerimientos de QA y sus prioridades relativas dependientes del
				  contexto de ejecución
				\end{itemize}

				En qué consiste la extensión a Rainbow (Rainbow + Escenarios)
			\end{frame}
