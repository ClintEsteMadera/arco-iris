\section{Descripción del Problema y Propuesta}

	\subsection{Mejoras a Rainbow}
	
		\begin{frame}
			\frametitle{\insertsubsection}
			Características a mejorar de Rainbow:
			\begin{itemize}
			  \item No involucra a los \emph{stakeholders} en el proceso de configuración de auto reparación
			  \item Poco dinámico, causas:
			  \begin{itemize}
			    \item configuración compleja
			    \item ausencia de mecanismo de actualización en caliente
			    \item no se adapta al entorno de ejecución
			   \end{itemize}
			  \item Duplicación de configuración de restricciones (\emph{overhead de procesamiento})
			\end{itemize}
		\end{frame}

	\subsection{Atributos de Calidad y Concerns}
		\begin{frame}[allowframebreaks,from second]
			\frametitle{\insertsubsection}
			Los \textbf{atributos de calidad} hablan de características específicas que debe tener el sistema 
			(anteriormente llamados ``requerimientos no funcionales'').\\[10pt]
			Ejemplo: eficiencia, disponibilidad, flexibilidad, usabilidad, etc.\\[16pt]
			
			Los \textbf{concerns} son parámetros mediante los cuales los atributos de un sistema son juzgados,
			especificados y medidos. Usualmente, los requerimientos de atributos de calidad son
			expresados en términos de concerns.
		
			\framebreak
			
			\frametitle{\insertsubsection}
			\begin{center}
				\rowcolors*[\hline]{1}{GreenYellow!25}{GreenYellow!10}
				\begin{tabular}{|c|c|}
				\textbf{Atributo de Calidad} & \textbf{\emph{Concerns}}\\
				Eficiencia & Comportamiento Temporal, Utilización de Recursos\\
				Funcionalidad & Interoperabilidad, Seguridad\\
				Mantenibilidad & Cambiabilidad, Facilidad de Prueba\\
				Portabilidad & Adaptabilidad, Coexistencia\\
				Usabilidad & Compresibilidad, Atractivo			
				\end{tabular}
			\end{center}
		\end{frame}

	\subsubsection{Escenarios de Atributos de Calidad}
		\begin{frame}[allowframebreaks,from second]
			\frametitle{\insertsubsubsection}
			Los Escenarios de Atributos de Calidad o QAS(\emph{Quality Attribute Scenarios}) permiten expresar cómo debería
			responder el sistema ante determinados estímulos y bajo ciertas circunstancias.\\
			
			\framebreak
			
			\frametitle{\insertsubsubsection}
			Composición de un QAS:

			\begin{figure}[ht]
				\centering \includegraphics[scale=0.8]{images/scenario.png}
			\end{figure}

			\nota{Estímulo: condición que debe ser tenida en cuenta cuando llega al sistema}
			\nota{Entorno: condiciones en las que se encuentra el sistema, e.g. alta carga}
			\nota{Artefacto: el sistema o partes de él afectadas por el estímulo}
			\nota{Respuesta: qué hace el sistema ante la llegada del estímulo}
			\nota{Cuantificación de la respuesta: métrica sobre la respuesta}

			La \textbf{Cuantificación de la respuesta} permitirá saber si el escenario se satisface.
		\end{frame}
	
	\subsubsection{Quality Attribute Workshop (QAW)}
		\begin{frame}
			\frametitle{\insertsubsubsection}
			Quality Attribute Workshop es un método que define un proceso que involucra a los \emph{stakeholders} en la
			definición de los Escenarios de Atributos de Calidad.
			
			Permite identificar drivers de la Arquitectura del sistema
		\end{frame}

	\subsection{Arco Iris}
	
		\subsubsection{¿Qué es Arco Iris?}
	
			\begin{frame}
				\frametitle{\insertsubsubsection}
				Arco Iris pretende involucrar a los \emph{stakeholders} en el proceso de auto reparación y proveerles
				una mayor visibilidad de sus posibles consecuencias para el desempeño del sistema.
	
				\begin{itemize}
				  \item Permite a stakeholders definir los requerimientos de QA y sus prioridades relativas dependientes del
				  contexto de ejecución
				\end{itemize}

				En qué consiste la extensión a Rainbow (Rainbow + Escenarios)
			\end{frame}
