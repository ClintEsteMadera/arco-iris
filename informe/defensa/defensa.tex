	\subsection{Atributos de Calidad y Concerns}
		\begin{frame}[allowframebreaks,from second]
			\frametitle{\insertsubsection}
			Los \textbf{atributos de calidad} (QAs) hablan de características específicas que debe tener el sistema, antes (mal)
			llamados ``requerimientos no funcionales''.\\[10pt]
			Ejemplo: eficiencia, disponibilidad, flexibilidad, usabilidad, etc.\\[16pt]

			\nota{ejemplo de definición QA: el sistema debe ser diseñado de manera tal que se minimize el procesamiento y el tiempo de respuesta.}

			Los \textbf{\emph{concerns}} permiten juzgar, especificar y medir los atributos de calidad. Usualmente, los requerimientos
			de atributos de calidad son expresados en términos de \emph{concerns}.
		
			\framebreak
			
			\frametitle{\insertsubsection}
			Ejemplos de \textbf{concerns}:
			\small
			\begin{center}
				\rowcolors*[\hline]{1}{GreenYellow!65}{GreenYellow!30}
				\begin{tabular}{|c|c|}
					\textbf{Atributo de Calidad} & \textbf{Ejemplos de \emph{Concerns}}\\
					Eficiencia & Comportamiento Temporal, Utilización de Recursos\\
					Funcionalidad & Interoperabilidad, Seguridad\\
					Mantenibilidad & Cambiabilidad, Facilidad de Prueba\\
					Portabilidad & Adaptabilidad, Coexistencia\\
					Usabilidad & Compresibilidad, Atractivo			
				\end{tabular}
			\end{center}
		\end{frame}

	\subsection{Rainbow}

		%--------- TURCO ---------
		\begin{frame}
			\frametitle{\insertsubsection}
			
			\nota{Adapt.Basada en modelos de Arq.: módulo q' conoce e interpreta el modelo de la arq. y de acuerdo al
			problema detectado, toma una decisión sobre cómo reparar al sistema}
			\nota{Intenta aplicar a arquitecturas disímiles y ser no intrusivo}
			\nota{No requiere diseñar el sistema teniendo en cuenta que va a ser adaptado mediante Rainbow}
			
			\begin{itemize}[<+->]
			  	\item \emph{Framework} de Auto Adaptación.
			  	\item Realizado bajo el marco del proyecto ABLE de la CMU. 
				\item Trabaja con el modelo de la arquitectura (expresado mediante \textbf{Acme}).
				\item Externo al sistema a adaptar.
				\item Soporta sistemas \emph{legacy} y/o cuyo código no se encuentra disponible.
			\end{itemize}
			
		\end{frame}

		%--------- TURCO ---------
		\begin{frame}[containsverbatim]
			\frametitle{Ejemplo de un Arquitectura en Acme}
			
			\nota{las invariantes seran utilizadas por Rainbow para saber cuándo y cómo debe adaptar el sistema}

			\scriptsize
			\begin{verbatim}
				Property MAX_RESPTIME : float = 1000.0;

				Component s1 : ServerT {
				  Property deploymentLocation;
				  Port http0;
				  Property cost;
				  Property fidelity;
				  Property load;
				}

				Component c1 : ClientT {
				  Property experRespTime : float << default : float = 100.0; >> ;
				  Property requestRate;
				  rule primaryConstraint = invariant self.experRespTime <= MAX_RESPTIME;
				}
			\end{verbatim}
		\end{frame}

		\begin{frame}[allowframebreaks]
			\frametitle{Tácticas y Estrategias de Reparación}

			\underline{\textbf{Táctica}}: decisión de diseño para controlar las respuestas del sistema a determinados
			estímulos, a fin de satisfacer uno o más QA.\\[5pt]
			e.g. \alert{agregar un servidor} al sistema.\\[14pt]
			\underline{\textbf{Estrategia}}: es un \textbf{algoritmo que usa una o más tácticas} para cumplir los
			atributos de calidad en un nivel deseado. 

			\framebreak
			\textbf{Invariante}: el tiempo de respuesta experimentado por el usuario no debe superar el umbral
			determinado.\\[10pt]
			
			\nota{explicar disminuir fidelidad de la información: por ejemplo eliminar videos}
			
			Estrategia que intenta reparar el sistema en caso de romperse el invariante:\\[5pt]
			\begin{enumerate}
			  \item \alert{agregar un servidor}
			  \item si tiempo de respuesta $>$ umbral determinado
			  \textbf{\textcolor{blue}{$\Rightarrow$}} \alert{disminuir la fidelidad}
			  \item Estrategia exitosa \textbf{\textcolor{blue}{$\iff$}} tiempo de respuesta $<$ umbral determinado  
			\end{enumerate}
		\end{frame}

	\subsection{Mejoras a Rainbow}
	
		\begin{frame}
			\frametitle{Características de Rainbow a Mejorar}

			\begin{itemize}[<+->]
				\item No involucra a los \emph{stakeholders} en el proceso de configuración de la auto reparación.
				\item Duplicación en la configuración de restricciones 
				\\\indent\textbf{\textcolor{blue}{\Large{$\Rightarrow$}}} \emph{overhead de procesamiento}
				\item Poco dinámico, debido a:
				\begin{itemize}[<+->]
					\item configuración compleja
					\item ausencia de mecanismo de actualización ``en caliente''
					\item no se adapta al entorno de ejecución
				\end{itemize}
			\end{itemize}
		\end{frame}
	
	\subsection{Escenarios de Atributos de Calidad}
	
		\subsubsection{Escenarios de Atributos de Calidad}
	
			\nota{Los Escenarios de Atributos de Calidad como forma fundamental de expresar el conocimiento de los
			\emph{stakeholders} referido a requerimientos que deben ser contemplados por la arquitectura}
	
			\begin{frame}[allowframebreaks,from second]
				\frametitle{\insertsubsubsection}
				Los \textbf{Escenarios de Atributos de Calidad} o QAS(\emph{Quality Attribute Scenarios}) permiten expresar cómo
				debería responder el sistema ante determinados estímulos y bajo ciertas circunstancias.
				
				\framebreak
				
				\frametitle{\insertsubsubsection}
				Composición de un QAS:

				\nota{Estímulo: condición que debe ser tenida en cuenta cuando llega al sistema}
				\nota{Entorno: condiciones en las que se encuentra el sistema, e.g. alta carga}
				\nota{Artefacto: el sistema o partes de él afectadas por el estímulo}
				\nota{Respuesta: qué hace el sistema ante la llegada del estímulo}

	\subsection{Arco Iris}
	
		\subsubsection{¿Qué es Arco Iris?}
	
			\begin{frame}[allowframebreaks,from second]
				\frametitle{\insertsubsubsection}
				
				Arco Iris pretende involucrar a los \emph{stakeholders} en el proceso de auto reparación y proveerles
				mayor visibilidad sobre la injerencia que este proceso tiene en el desempeño del sistema.

				\framebreak

				\begin{figure}[ht]
					\centering \includegraphics[scale=0.4]{images/arcoIrisUI_conceptualFlow.png}
				\end{figure}

				\framebreak

				Características:
				\begin{itemize}
				  \item Utiliza los QAS definidos por los \emph{stakeholders} en los QAW para llevar a cabo la auto reparación
				  \item Extiende los QAS con información requerida para mejorar la adaptación:
				  	\begin{itemize}
				    	\item Cada escenario tiene una prioridad determinada
				    	\item Cada entorno define la importancia de los distintos \emph{concerns}
				    	\item Estrategias que reparan al escenario
					\end{itemize}
				\end{itemize}

				\begin{center}
					Arco Iris = Rainbow + Escenarios
				\end{center}
			\end{frame}
