Mappings para Hibernate

Persona.hbm.xml

			<?xml version="1.0"?>
			<!DOCTYPE hibernate-mapping PUBLIC
				"-//Hibernate/Hibernate Mapping DTD 3.0//EN"
				"http://hibernate.sourceforge.net/hibernate-mapping-3.0.dtd">

			<hibernate-mapping default-access="field" package="sba.common.persona.core.domain">

				<class abstract="true" name="Persona" table="PERSONA">

					<id name="id">
						<generator class="sequence"/>
					</id>
					<discriminator column="TIPO_PERSONA" type="string" length="20"/>
					<version name="version" column="VERSION_NUMBER"/>

				</class>

			</hibernate-mapping>

PersonaFisica.hbm.xml

		<?xml version="1.0"?>
		<!DOCTYPE hibernate-mapping PUBLIC
			"-//Hibernate/Hibernate Mapping DTD 3.0//EN"
			"http://hibernate.sourceforge.net/hibernate-mapping-3.0.dtd">

		<hibernate-mapping default-access="field" package="sba.common.persona.core.domain">

			<subclass name="PersonaFisica"
					extends="Persona" discriminator-value="PERSONA_FISICA">
				
				<property name="nombre" length="50"/>
				<property name="apellido" length="50"/>
				
				<component name="dni" class="sba.common.persona.identificacion.domain.NroDNI"> 
						<property name="numero" column="DNI"/>
				</component> 
				
			
			</subclass>

		</hibernate-mapping>

PersonaJuridica.hbm.xml

		<?xml version="1.0"?>
		<!DOCTYPE hibernate-mapping PUBLIC
			"-//Hibernate/Hibernate Mapping DTD 3.0//EN"
			"http://hibernate.sourceforge.net/hibernate-mapping-3.0.dtd">

		<hibernate-mapping default-access="field" package="sba.common.persona.core.domain">

			<subclass name="PersonaJuridica" 
				extends="Persona" discriminator-value="PERSONA_JURIDICA">
				
				<property name="denominacion" unique="true" length="50"/>
				
				<component name="cuit" class="sba.common.persona.identificacion.domain.NroCUIT"> 
						<property name="numero" column="CUIT"/>
				</component> 
				
			</subclass>

		</hibernate-mapping>
